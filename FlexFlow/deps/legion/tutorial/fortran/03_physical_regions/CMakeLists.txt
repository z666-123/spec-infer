cmake_minimum_required(VERSION 3.16 FATAL_ERROR)
project(LegionExample_05_physical_regions_fortran)

# Only search if were building stand-alone and not as part of Legion
if(NOT Legion_SOURCE_DIR)
  find_package(Legion REQUIRED)
endif()

add_executable(physical_regions_fortran physical_regions.f90)
target_link_libraries(physical_regions_fortran Legion::Legion)
if(Legion_ENABLE_TESTING)
  add_test(NAME physical_regions_fortran COMMAND ${Legion_TEST_LAUNCHER} $<TARGET_FILE:physical_regions_fortran> ${Legion_TEST_ARGS})
endif()
