cmake_minimum_required(VERSION 3.16 FATAL_ERROR)
project(LegionExample_06_privileges_accessor_fortran)

# Only search if were building stand-alone and not as part of Legion
if(NOT Legion_SOURCE_DIR)
  find_package(Legion REQUIRED)
endif()

add_executable(privileges_accessor_fortran privileges.f90)
target_link_libraries(privileges_accessor_fortran Legion::Legion)
if(Legion_ENABLE_TESTING)
  add_test(NAME privileges_accessor_fortran COMMAND ${Legion_TEST_LAUNCHER} $<TARGET_FILE:privileges_accessor_fortran> ${Legion_TEST_ARGS})
endif()
