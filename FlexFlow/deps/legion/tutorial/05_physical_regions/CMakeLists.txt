cmake_minimum_required(VERSION 3.16 FATAL_ERROR)
project(LegionExample_05_physical_regions)

# Only search if were building stand-alone and not as part of Legion
if(NOT Legion_SOURCE_DIR)
  find_package(Legion REQUIRED)
endif()

add_executable(physical_regions physical_regions.cc)
target_link_libraries(physical_regions Legion::Legion)
target_compile_options(physical_regions PRIVATE $<$<COMPILE_LANGUAGE:CXX>:${CXX_BUILD_WARNING_FLAGS}>)
if(Legion_ENABLE_TESTING)
  add_test(NAME physical_regions COMMAND ${Legion_TEST_LAUNCHER} $<TARGET_FILE:physical_regions> ${Legion_TEST_ARGS})
endif()
